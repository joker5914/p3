# =============================================================================
# Dismissal Scanner — Raspberry Pi 5 + Arducam 16MP (CSI)
# =============================================================================
#
# Install with:
#   pip install -r requirements-scanner.txt
#
# IMPORTANT — these packages MUST come from apt, NOT pip:
#   sudo apt-get install -y \
#       python3-picamera2 python3-libcamera python3-kms++ \
#       tesseract-ocr tesseract-ocr-eng
#
# The Python venv MUST be created with --system-site-packages so that
# picamera2 (an apt package) is visible inside the venv:
#   python3 -m venv --system-site-packages /opt/dismissal/venv
#
# The install script (deploy/install.sh) handles all of this automatically.
#
# Plate detection uses tflite-runtime — runs on the Pi 5 CPU and can
# optionally use a Google Coral Edge TPU via its libedgetpu.so delegate
# when one is plugged in.  pycoral is NOT used (abandoned on Python 3.10+).
# =============================================================================

# --- Computer vision ---
opencv-python-headless==4.11.0.86   # headless build — no GUI/display needed
numpy==2.1.3

# --- OCR ---
pytesseract==0.3.13
Pillow==12.2.0

# --- HTTP client ---
requests==2.33.0

# --- Environment config ---
python-dotenv==1.2.2

# --- Firebase auth (minting + refreshing ID tokens from a service-account JSON) ---
firebase-admin==6.5.0

# --- Neural vehicle detector (CPU or Coral Edge TPU) ---
# Installed best-effort by deploy/install.sh (tries tflite-runtime first,
# then ai-edge-litert).  Not pinned here because wheel availability on
# Trixie / Python 3.13 is still in flux, and PlateDetector falls back
# to contour-only plate detection when the import fails.

# --- Direct plate detector (YOLOv8 ONNX) ---
# Runs a license-plate-trained YOLO via ONNX Runtime for close-up
# detection where the full car isn't visible.  deploy/install_plate_model.sh
# downloads the actual .onnx model file — this line only installs the
# runtime.
onnxruntime

# --- Plate-trained OCR (CRNN via fast-plate-ocr) ---
# Much more accurate than Tesseract on real-world US plates.  First
# use downloads the model (~10 MB) into ~/.cache/fast_plate_ocr.
# Tesseract stays installed as a fallback when the import or first-run
# download fails.
fast-plate-ocr

# --- Power-button factory-reset daemon ---
# Reads /dev/input/event* for KEY_POWER taps to detect the multi-tap
# factory-reset gesture (5 presses within 3 seconds).  python3-evdev is
# also available as an apt package; we install via pip inside the venv
# so the import path matches the rest of the scanner.
evdev

# python-libcamera apt: python3-libcamera

# --- NOT included (backend-only, do not install on scanner RPi) ---
